331 void CCStepperDevice_TMC260::calculateCurrentSetup(unsigned int current) {
332  // the mesurement full scale on the chip can eiter be set to 165mV or 310mV
333  // the shunt on the shield (R_SENSE) is 150mOhm
334  // VsenseMax = R_SENSE * coilCurrentMax [/ 1000]
335  // ==> coilCurrentMax = VsenseMax / R_SENSE:
336  // VsenseMax = 165mV ==> coilCurrentMax = 165mV / 150mOhm = 1100mA
337  // VsenseMax = 310mV ==> coilCurrentMax = 310mV / 150mOhm = 2066mA
338 
339  unsigned int VSenseFullScale; // voltage in mV - it can either be 165mV or 310MV
340  if ((long) R_SENSE * current <= 165000) {
341  VSenseFullScale = 165;
342  senseResistorVoltage165mV = true;
343  }
344  else {
345  VSenseFullScale = 310;
346  senseResistorVoltage165mV = false;
347  }
348  // from datasheet: I = (CS + 1) / 32 * Vfs / Rs
349  // here: CS + 1 = currentScaleOf32; Vfs = VSenseFullScale / 1000; Rs = R_SENSE / 1000
350  // so: I / 1000 = currentScaleOf32 / 32 * VSenseFullScale / R_SENSE
351  // ==> currentScaleOf32 = (32 * R_SENSE * I) / VSenseFullScale;
352  currentScaleOf32 = (float)(32.0 * R_SENSE * current) / VSenseFullScale / 1000.0 + 0.5; // (+ 0.5 is for better rounding)
353  // 300 mA: 32 * 0.15 * 0.3 / 0.165 = 8.73 --> 9
354  // 600 mA: 32 * 0.15 * 0.6 / 0.165 = 17.45 --> 17
355  // 1000 mA: 32 * 0.15 * 1.0 / 0.165 = 29.09 --> 29
356  // 1100 mA: 32 * 0.15 * 1.1 / 0.165 = 32 --> 32
357  // 1200 mA: 32 * 0.15 * 1.1 / 0.310 = 18.58 --> 19
358  // 2060 mA: 32 * 0.15 * 1.0 / 0.310 = 31.89 --> 32
359 
360  currentScaleOf32 = min(currentScaleOf32, 32);
361 
362  if (STEPPER_TMC260_VERBOSE & TMC260_SETUPDEBUG) {
363  Serial.print(F("[CCStepperDevice_TMC260]: "));
364  Serial.print(deviceName);
365  if (senseResistorVoltage165mV) {
366  Serial.print(F(": currentScaling full scale, "));
367  } else {
368  Serial.print(F(": currentScaling halfed, "));
369  }
370  Serial.print(F(" currentScaleOf32: "));
371  Serial.println(currentScaleOf32);
372  }
