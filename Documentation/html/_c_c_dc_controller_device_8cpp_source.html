221 void CCDcControllerDevice::operateTask() {
222  elapsedTime = millis() - t0;
223 
224  if (elapsedTime >= switchOnTime) {
225 
226  if (DCCONTROLLER_VERBOSE & MOVEMENTDEBUG) {
227  Serial.print(elapsedTime);
228  Serial.print(F(": switch on"));
229  }
230 
231  if (elapsedTime < timeForAcceleration) {
232  currentPosition = currentRatio + (target - currentRatio) * elapsedTime / (float)timeForAcceleration;
233 
234  if (DCCONTROLLER_VERBOSE & MOVEMENTDEBUG) {
235  Serial.print(F(", accelerating"));
236  }
237 
238  } else if (elapsedTime < timeForAccAndConstSpeed) {
239  currentPosition = target;
240 
241  if (DCCONTROLLER_VERBOSE & MOVEMENTDEBUG) {
242  Serial.print(F(", runing regular"));
243  }
244  } else {
245  elapsedTime -= timeForAccAndConstSpeed;
246  if (elapsedTime < timeForDeceleration) {
247  currentPosition = target * (1 - elapsedTime / (float)timeForDeceleration);
248 
249  if (DCCONTROLLER_VERBOSE & MOVEMENTDEBUG) {
250  Serial.print(F(", decelerating"));
251  }
252 
253  } else {
254 
255  if (DCCONTROLLER_VERBOSE & MOVEMENTDEBUG) {
256  Serial.print(F(", shall stop"));
257  }
258 
259  stopTask();
260  return;
261  }
262  }
263 
264  if (stopping == STOP_DYNAMIC) {
265  sensorValue = analogRead(sensor);
266  if (reversedApproximation) {
267  relativePosition = targetValue - sensorValue;
268  } else {
269  relativePosition = sensorValue - targetValue;
270  }
271 
272  if (DCCONTROLLER_VERBOSE & MOVEMENTDEBUG) {
273  Serial.print(F(", stopping dynamically"));
274  }
275 
276  if (relativePosition <= gap) {
277 
278  currentPosition = 0;
279  targetReachedCounter++;
280 
281  if (DCCONTROLLER_VERBOSE & MOVEMENTDEBUG) {
282  Serial.print(F(", times target was reached: "));
283  Serial.print(targetReachedCounter);
284  }
285 
286  if (targetReachedCounter >= approximation) {
287  stopTask();
288  }
289  return;
290 
291  } else {
292 
293  // equitation first order:
294  // y = -b / (x - a) + 1; (1)
295  //
296  // crossPoint: y = 0 ==>
297  // 0 = -b / (x - a) + 1; b / (x - a) = 1;
298  // b = x - a; x = a + b;
299  // crossPoint at x = a + b: cp = a + b ==> a = cp - b
300  //
301  // in 1:
302  // y = -b / (x - cp + b) + 1 = b / (cp - x - b) + 1
303 
304  // currentPosition *= approximationCurve / (gap * gap - relativePosition * relativePosition - approximationCurve) + 1;
305  currentPosition *= approximationCurve / (gap - relativePosition - approximationCurve) + 1;
306 
307  targetReachedCounter = 0;
308 
309  }
310  }
311 
312  digitalWrite(switching_pin, switchingPin_activ);
313  isActiv = true;
314 
315  switchOffTime = switchOnTime + switchingInterval * currentPosition;
316  switchOnTime += switchingInterval;
317  if (DCCONTROLLER_VERBOSE & MOVEMENTDEBUG) {
318  Serial.print(F(", scheduled off at "));
319  Serial.print(switchOffTime);
320  Serial.print(F(", duty cycle: "));
321  Serial.print(currentPosition);
322  Serial.print(F(", next cycle at "));
323  Serial.println(switchOnTime);
324  }
325 
326 
327  } else if (elapsedTime > switchOffTime) {
328  if (isActiv == true) {
329  digitalWrite(switching_pin, !switchingPin_activ);
330  isActiv = false;
331  if (DCCONTROLLER_VERBOSE & MOVEMENTDEBUG) {
332  Serial.print(elapsedTime);
333  Serial.print(F(": switched off at "));
334  Serial.println(switchOffTime);
335  }
336  }
337  }
