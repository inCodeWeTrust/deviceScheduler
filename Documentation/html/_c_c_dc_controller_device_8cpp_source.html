194 void CCDcControllerDevice::operateTask() {
195  elapsedTime = millis() - t0;
196 
197  if (elapsedTime >= switchOnTime) {
198 
199  if (verbosity & MOVEMENTDEBUG) {
200  Serial.print(elapsedTime);
201  Serial.print(F(": switch on"));
202  }
203 
204  if (elapsedTime < timeForAcceleration) {
205  currentPosition = currentRatio + (target - currentRatio) * elapsedTime / (float)timeForAcceleration;
206 
207  if (verbosity & MOVEMENTDEBUG) {
208  Serial.print(F(", accelerating"));
209  }
210 
211  } else if (elapsedTime < timeForAccAndConstSpeed) {
212  currentPosition = target;
213 
214  if (verbosity & MOVEMENTDEBUG) {
215  Serial.print(F(", runing regular"));
216  }
217  } else {
218  elapsedTime -= timeForAccAndConstSpeed;
219  if (elapsedTime < timeForDeceleration) {
220  currentPosition = target * (1 - elapsedTime / (float)timeForDeceleration);
221 
222  if (verbosity & MOVEMENTDEBUG) {
223  Serial.print(F(", decelerating"));
224  }
225 
226  } else {
227 
228  if (verbosity & MOVEMENTDEBUG) {
229  Serial.print(F(", shall stop"));
230  }
231 
232  stopTask();
233  return;
234  }
235  }
236 
237  if (stopping == STOP_DYNAMIC) {
238  sensorValue = approximationControl->value();
239  if (reversedApproximation) {
240  relativePosition = approximationTarget - sensorValue;
241  } else {
242  relativePosition = sensorValue - approximationTarget;
243  }
244 
245  if (verbosity & MOVEMENTDEBUG) {
246  Serial.print(F(", stopping dynamically"));
247  }
248 
249  if (relativePosition <= gap) {
250 
251  currentPosition = 0;
252  targetReachedCounter++;
253 
254  if (verbosity & MOVEMENTDEBUG) {
255  Serial.print(F(", times target was reached: "));
256  Serial.print(targetReachedCounter);
257  }
258 
259  if (targetReachedCounter >= approximation) {
260  stopTask();
261  }
262  return;
263 
264  } else {
265 
266  // equitation first order:
267  // y = -b / (x - a) + 1; (1)
268  //
269  // crossPoint: y = 0 ==>
270  // 0 = -b / (x - a) + 1; b / (x - a) = 1;
271  // b = x - a; x = a + b;
272  // crossPoint at x = a + b: cp = a + b ==> a = cp - b
273  //
274  // in 1:
275  // y = -b / (x - cp + b) + 1 = b / (cp - x - b) + 1
276 
277  // currentPosition *= approximationCurve / (gap * gap - relativePosition * relativePosition - approximationCurve) + 1;
278  // currentPosition *= approximationCurve / (gap - relativePosition - approximationCurve) + 1;
279 
280  currentPosition = target * (approximationCurve / (gap * gap / 16.0 - relativePosition * relativePosition - approximationCurve) + 1);
281 
282  targetReachedCounter = 0;
283 
284  }
285  }
286 
287  digitalWrite(switching_pin, switchingPin_active);
288  isActive = true;
289 
290  switchOffTime = switchOnTime + switchingInterval * currentPosition;
291  switchOnTime += switchingInterval;
292  if (verbosity & MOVEMENTDEBUG) {
293  Serial.print(F(", scheduled off at "));
294  Serial.print(switchOffTime);
295  Serial.print(F(", duty cycle: "));
296  Serial.print(currentPosition);
297  Serial.print(F(", next cycle at "));
298  Serial.println(switchOnTime);
299  }
300 
301 
302  } else if (elapsedTime > switchOffTime) {
303  if (isActive == true) {
304  digitalWrite(switching_pin, !switchingPin_active);
305  isActive = false;
306  if (verbosity & MOVEMENTDEBUG) {
307  Serial.print(elapsedTime);
308  Serial.print(F(": switched off at "));
309  Serial.println(switchOffTime);
310  }
311  }
312  }
