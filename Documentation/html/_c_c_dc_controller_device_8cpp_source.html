187 void CCDcControllerDevice::operateTask() {
188  elapsedTime = millis() - t0;
189 
190  if (elapsedTime >= switchOnTime) {
191  digitalWrite(switching_pin, switchingPin_activ);
192  isActiv = true;
193  if (elapsedTime < timeForAcceleration) {
194  currentPosition = currentRatio + (target - currentRatio) * elapsedTime / (float)timeForAcceleration;
195  switchOffTime = switchOnTime + switchingInterval * currentPosition;
196  switchOnTime += switchingInterval;
197  if (CCDcControllerDevice_VERBOSE & CCDcControllerDevice_MOVEMENTDEBUG) {
198  Serial.print(elapsedTime);
199  Serial.print(F(": switched on: accellerating, switch off at "));
200  Serial.print(switchOffTime);
201  Serial.print(F(", duty cycle: "));
202  Serial.print(currentPosition);
203  Serial.print(F(", next cycle at "));
204  Serial.println(switchOnTime);
205  }
206  return;
207  } else if (elapsedTime < timeForAccAndConstSpeed) {
208  currentPosition = target;
209  switchOffTime = switchOnTime + targetOnDuration;
210  switchOnTime += switchingInterval;
211  if (CCDcControllerDevice_VERBOSE & CCDcControllerDevice_MOVEMENTDEBUG) {
212  Serial.print(elapsedTime);
213  Serial.print(F(": switched on, switch off at "));
214  Serial.print(switchOffTime);
215  Serial.print(F(", duty cycle: "));
216  Serial.print(currentPosition);
217  Serial.print(F(", next cycle at "));
218  Serial.println(switchOnTime);
219  }
220  return;
221  } else {
222  elapsedTime -= timeForAccAndConstSpeed;
223  if (elapsedTime < timeForDeceleration) {
224  currentPosition = target * (1 - elapsedTime / (float)timeForDeceleration);
225  switchOffTime = switchOnTime + switchingInterval * currentPosition;
226  switchOnTime += switchingInterval;
227  if (CCDcControllerDevice_VERBOSE & CCDcControllerDevice_MOVEMENTDEBUG) {
228  Serial.print(elapsedTime);
229  Serial.print(F(": switched on: decellerating, switch off at "));
230  Serial.print(switchOffTime);
231  Serial.print(F(", duty cycle: "));
232  Serial.print(currentPosition);
233  Serial.print(F(", next cycle at "));
234  Serial.println(switchOnTime);
235  }
236  return;
237  }
238  // if we didnt return up to here, we are done!
239  stopTask();
240  }
241 
242  } else if (elapsedTime > switchOffTime) {
243  if (isActiv == true) {
244  digitalWrite(switching_pin, !switchingPin_activ);
245  isActiv = false;
246  if (CCDcControllerDevice_VERBOSE & CCDcControllerDevice_MOVEMENTDEBUG) {
247  Serial.print(elapsedTime);
248  Serial.print(F(": switched off at "));
249  Serial.println(switchOffTime);
250  }
251  }
252  }
