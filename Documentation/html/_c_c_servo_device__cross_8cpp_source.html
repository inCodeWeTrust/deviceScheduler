116 void CCServoDevice_cross::operateTask() {
117  lastCycleTime = elapsedTime;
118  elapsedTime = millis() - t0;
119 
120  if (stopping == STOP_DYNAMIC) {
121  if (dynamicalStop == false) {
122  sensorValue = analogRead(sensor);
123  if ((sensorValue > initiatePerformanceValue && (!sensorValuesFalling)) || (sensorValue < initiatePerformanceValue && sensorValuesFalling)) {
124  timeForAcceleration = elapsedTime;
125  timeForConstantSpeed = 0;
126  lastCycleTime = 0;
127  c_perform = 1.0 / (initiatePerformanceValue - targetValue);
128  dynamicalStop = true;
129  }
130  }
131  }
132 
133  // ramp up
134  if (elapsedTime < (signed long)timeForAcceleration) {
135  // s = s0 + 0.5 * a * t^2 * (1000ms / sec)^2
136  deltaS = elapsedTime * elapsedTime * acceleration / 2000000;
137  currentPosition = startPosition + deltaS;
138  theServo.writeMicroseconds(currentPosition);
139  secondServo.writeMicroseconds(counterPosition(currentPosition));
140 
141  if (CCSERVODEVICE_CROSS_VERBOSE & CCSERVODEVICE_CROSS_MOVEMENTDEBUG) {
142  Serial.print(F("[CCServoDevice_cross]: "));
143  Serial.print(deviceName);
144  Serial.print(F(": ramp up: cur: "));
145  Serial.print((int)currentPosition);
146  Serial.print(F(", start: "));
147  Serial.print((int)startPosition);
148  Serial.print(F(", deltaS: "));
149  Serial.println((long)deltaS);
150  }
151 
152  return;
153  }
154 
155  // go with constant speed
156  elapsedTime -= timeForAcceleration;
157  if (elapsedTime < (signed long)timeForConstantSpeed) {
158  // s = s0 + s1 + deltaS = s0 + s1 + v * t
159  deltaS = elapsedTime * velocity / 1000;
160  currentPosition = startPosition + wayForAcceleration + deltaS;
161  theServo.writeMicroseconds(currentPosition);
162  secondServo.writeMicroseconds(counterPosition(currentPosition));
163 
164  if (CCSERVODEVICE_CROSS_VERBOSE & CCSERVODEVICE_CROSS_MOVEMENTDEBUG) {
165  Serial.print(F("[CCServoDevice_cross]: "));
166  Serial.print(deviceName);
167  Serial.print(F(": go constant: cur: "));
168  Serial.print((int)currentPosition);
169  Serial.print(F(", start: "));
170  Serial.print((int)startPosition);
171  Serial.print(F(", wayAcc: "));
172  Serial.print((int)wayForAcceleration);
173  Serial.print(F(", deltaS: "));
174  Serial.println((long)deltaS);
175  }
176 
177  return;
178  }
179  // stop dynamically?
180  if (dynamicalStop) {
181  deltaDeltaNorm = (float)(elapsedTime - lastCycleTime) * velocity / 1000.0;
182 
183  sensorValue = analogRead(sensor);
184 
185  performanceFactor = c_perform * (sensorValue - targetValue);
186 
187  currentPosition += deltaDeltaNorm * performanceFactor;
188  currentPosition = min(currentPosition, maxPosition);
189  currentPosition = max(currentPosition, minPosition);
190 
191  theServo.writeMicroseconds(currentPosition);
192  secondServo.writeMicroseconds(counterPosition(currentPosition));
193 
194  if (CCSERVODEVICE_CROSS_VERBOSE & CCSERVODEVICE_CROSS_MOVEMENTDEBUG) {
195  Serial.print("elapsedTime: ");
196  Serial.print(elapsedTime);
197  Serial.print(", performanceFactor: ");
198  Serial.print(performanceFactor);
199  Serial.print(", sensorValue: ");
200  Serial.print(sensorValue);
201  Serial.print(", deltaDeltaNorm: ");
202  Serial.print(deltaDeltaNorm);
203  Serial.print(", deltaS: ");
204  Serial.print(deltaS);
205  Serial.print(", currentPosition: ");
206  Serial.println(currentPosition);
207  }
208 
209 
210  if (approximation == SKIP_APPROXIMATION_NEVER) {
211  return;
212  } else {
213  if (abs(sensorValue - targetValue) < sensorTreshold) {
214  if (valueCounter++ < approximation) {
215  return;
216  }
217  } else {
218  valueCounter = 0;
219  return;
220  }
221  }
222 
223 
224  if (CCSERVODEVICE_CROSS_VERBOSE & CCSERVODEVICE_CROSS_BASICOUTPUT) {
225  Serial.print(F("[CCServoDevice_cross]: "));
226  Serial.print(deviceName);
227  Serial.print(F(": stopped at sensor: "));
228  Serial.println(sensorValue);
229  }
230  targetPosition = currentPosition;
231 
232  } else {
233  // generic ramp down
234  elapsedTime -= timeForConstantSpeed;
235  if (elapsedTime < (signed long)timeForAcceleration) {
236  // t^2 = (timeForAcceleration - elapsedTime)^2 = (elapsedTime - timeForAcceleration)^2;
237  elapsedTime -= timeForAcceleration;
238  // s = sges - deltaS = 0.5 * a * t^2 = 0.5 * a * (time_in_millis / 1000)^2 = 1/2 * 1/1000000 * a * t^2
239  deltaS = elapsedTime * elapsedTime * acceleration / 2000000;
240  currentPosition = targetPosition - deltaS;
241  theServo.writeMicroseconds(currentPosition);
242  secondServo.writeMicroseconds(counterPosition(currentPosition));
243 
244  if (CCSERVODEVICE_CROSS_VERBOSE & CCSERVODEVICE_CROSS_MOVEMENTDEBUG) {
245  Serial.print(F("[CCServoDevice_cross]: "));
246  Serial.print(deviceName);
247  Serial.print(F(": ramp down: cur: "));
248  Serial.print((int)currentPosition);
249  Serial.print(F(", target: "));
250  Serial.print((int)targetPosition);
251  Serial.print(F(", deltaS: "));
252  Serial.println((long)deltaS);
253  }
254 
255  return;
256  }
257  }
258 
259 
260  // if we didnt return up to here, we are done!
261 
262  theServo.writeMicroseconds(targetPosition);
263  secondServo.writeMicroseconds(counterPosition(currentPosition));
264  currentPosition = targetPosition;
265 
266  stopTask();
267
