84 void CCServoDevice_cross::operateTask() {
85  lastCycleTime = elapsedTime;
86  elapsedTime = millis() - t0;
87 
88  if (stopping == STOP_DYNAMIC) {
89  if (dynamicalStop == false) {
90  sensorValue = approximationControl->value();
91  if ((sensorValue > initiatePerformanceValue && (!sensorValuesFalling)) || (sensorValue < initiatePerformanceValue && sensorValuesFalling)) {
92  timeForAcceleration = elapsedTime;
93  timeForConstantSpeed = 0;
94  lastCycleTime = 0;
95  c_perform = 1.0 / (initiatePerformanceValue - approximationTarget);
96  dynamicalStop = true;
97  }
98  }
99  }
100 
101  // ramp up
102  if (elapsedTime < (signed long)timeForAcceleration) {
103  // s = s0 + 0.5 * a * t^2 * (1000ms / sec)^2
104  deltaS = elapsedTime * elapsedTime * acceleration / 2000000;
105  currentPosition = startPosition + deltaS;
106  theServo.writeMicroseconds(currentPosition);
107  secondServo.writeMicroseconds(counterPosition(currentPosition));
108 
109  if (verbosity & MOVEMENTDEBUG) {
110  Serial.print(F("[CCServoDevice_cross]: "));
111  Serial.print(deviceName);
112  Serial.print(F(": ramp up: cur: "));
113  Serial.print((int)currentPosition);
114  Serial.print(F(", start: "));
115  Serial.print((int)startPosition);
116  Serial.print(F(", deltaS: "));
117  Serial.println((long)deltaS);
118  }
119 
120  return;
121  }
122 
123  // go with constant speed
124  elapsedTime -= timeForAcceleration;
125  if (elapsedTime < (signed long)timeForConstantSpeed) {
126  // s = s0 + s1 + deltaS = s0 + s1 + v * t
127  deltaS = elapsedTime * velocity / 1000;
128  currentPosition = startPosition + wayForAcceleration + deltaS;
129  theServo.writeMicroseconds(currentPosition);
130  secondServo.writeMicroseconds(counterPosition(currentPosition));
131 
132  if (verbosity & MOVEMENTDEBUG) {
133  Serial.print(F("[CCServoDevice_cross]: "));
134  Serial.print(deviceName);
135  Serial.print(F(": go constant: cur: "));
136  Serial.print((int)currentPosition);
137  Serial.print(F(", start: "));
138  Serial.print((int)startPosition);
139  Serial.print(F(", wayAcc: "));
140  Serial.print((int)wayForAcceleration);
141  Serial.print(F(", deltaS: "));
142  Serial.println((long)deltaS);
143  }
144 
145  return;
146  }
147  // stop dynamically?
148  if (dynamicalStop) {
149  deltaDeltaNorm = (float)(elapsedTime - lastCycleTime) * velocity / 1000.0;
150 
151  sensorValue = approximationControl->value();
152 
153  performanceFactor = c_perform * (sensorValue - approximationTarget);
154 
155  currentPosition += deltaDeltaNorm * performanceFactor;
156  currentPosition = fmin(currentPosition, maxPosition);
157  currentPosition = fmax(currentPosition, minPosition);
158 
159  theServo.writeMicroseconds(currentPosition);
160  secondServo.writeMicroseconds(counterPosition(currentPosition));
161 
162  if (verbosity & MOVEMENTDEBUG) {
163  Serial.print("elapsedTime: ");
164  Serial.print(elapsedTime);
165  Serial.print(", performanceFactor: ");
166  Serial.print(performanceFactor);
167  Serial.print(", sensorValue: ");
168  Serial.print(sensorValue);
169  Serial.print(", deltaDeltaNorm: ");
170  Serial.print(deltaDeltaNorm);
171  Serial.print(", deltaS: ");
172  Serial.print(deltaS);
173  Serial.print(", currentPosition: ");
174  Serial.println(currentPosition);
175  }
176 
177 
178  if (approximation == SKIP_APPROXIMATION_NEVER) {
179  return;
180  } else {
181  if (fabs(sensorValue - approximationTarget) < sensorTreshold) {
182  if (valueCounter++ < approximation) {
183  return;
184  }
185  } else {
186  valueCounter = 0;
187  return;
188  }
189  }
190 
191 
192  if (verbosity & BASICOUTPUT) {
193  Serial.print(F("[CCServoDevice_cross]: "));
194  Serial.print(deviceName);
195  Serial.print(F(": stopped at sensor: "));
196  Serial.println(sensorValue);
197  }
198  targetPosition = currentPosition;
199 
200  } else {
201  // generic ramp down
202  elapsedTime -= timeForConstantSpeed;
203  if (elapsedTime < (signed long)timeForAcceleration) {
204  // t^2 = (timeForAcceleration - elapsedTime)^2 = (elapsedTime - timeForAcceleration)^2;
205  elapsedTime -= timeForAcceleration;
206  // s = sges - deltaS = 0.5 * a * t^2 = 0.5 * a * (time_in_millis / 1000)^2 = 1/2 * 1/1000000 * a * t^2
207  deltaS = elapsedTime * elapsedTime * acceleration / 2000000;
208  currentPosition = targetPosition - deltaS;
209  theServo.writeMicroseconds(currentPosition);
210  secondServo.writeMicroseconds(counterPosition(currentPosition));
211 
212  if (verbosity & MOVEMENTDEBUG) {
213  Serial.print(F("[CCServoDevice_cross]: "));
214  Serial.print(deviceName);
215  Serial.print(F(": ramp down: cur: "));
216  Serial.print((int)currentPosition);
217  Serial.print(F(", target: "));
218  Serial.print((int)targetPosition);
219  Serial.print(F(", deltaS: "));
220  Serial.println((long)deltaS);
221  }
222 
223  return;
224  }
225  }
226 
227 
228  // if we didnt return up to here, we are done!
229 
230  theServo.writeMicroseconds(targetPosition);
231  secondServo.writeMicroseconds(counterPosition(currentPosition));
232  currentPosition = targetPosition;
233 
234  stopTask();
235
