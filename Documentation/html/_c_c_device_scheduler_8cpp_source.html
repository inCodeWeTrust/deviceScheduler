296 int CCDeviceScheduler::run(CCWorkflow* currentWorkflow) {
297 
298  unsigned long loopCounter = 0;
299  unsigned char ongoingOperations = 0;
300  boolean breakLoop = false;
301  notificationCode = 0;
302  notificationText = "";
303 
304 
305  if ((char)currentWorkflow->getWorkflowInfo() >= 0) {
306 
307  // prepare the loop
308 
309  if (SCHEDULER_VERBOSE & SHOW_TASK_VIEW) {
310  Serial.print(F("[CCDeviceScheduler]: "));
311  Serial.print(schedulerName);
312  Serial.println(F(": prepare run ..."));
313  }
314  for (unsigned char df = 0; df < currentWorkflow->getCountOfDeviceFlows(); df++) {
315  currentDeviceFlow = currentWorkflow->deviceFlow[df];
316  currentDeviceFlow->setTaskPointer(0);
317 
318  currentDevice = currentDeviceFlow->device;
319  currentDevice->setState(PENDING_MOVES);
320  currentDevice->attachDevice();
321 
322  // prepare first Tasks
323  currentDevice->prepareTask(currentDeviceFlow->task[0]);
324 
325 
326  if (SCHEDULER_VERBOSE & SHOW_TASK_VIEW) {
327  Serial.print(currentWorkflow->getName());
328  Serial.print(F("->"));
329  Serial.print(currentDeviceFlow->getName());
330  Serial.print(F(": setup first Task: current Position: "));
331  Serial.print(currentDevice->getCurrentPosition());
332  Serial.print(F(", target: "));
333  Serial.println(currentDevice->getTarget());
334  }
335  delay(200);
336  }
337 
338  for (unsigned char fc = 0; fc < currentWorkflow->getCountOfFlowControls(); fc++) {
339  currentWorkflow->flowControl[fc]->button->readButtonState();
340  }
341 
342 
343  // start the workflow
344  if (SCHEDULER_VERBOSE & SHOW_TASK_VIEW) {
345  Serial.print(F("[CCDeviceScheduler]: "));
346  Serial.print(schedulerName);
347  Serial.print(F(": run workflow "));
348  Serial.println(currentWorkflow->getName());
349  }
350 
351  unsigned long taskTime, loopTime;
352  unsigned long taskStartTime = millis();
353 
354  // unsigned long lastPrintTime = taskStartTime;
355 
356  schedulerControlButton theButtonID = 0;
357 
358  do {
359  loopTime = millis();
360  taskTime = loopTime - taskStartTime;
361 
362  ongoingOperations = 0;
363 
364 
365  for (unsigned char df = 0; df < currentWorkflow->getCountOfDeviceFlows(); df++) {
366  currentDeviceFlow = currentWorkflow->deviceFlow[df];
367  currentDevice = currentDeviceFlow->device;
368  currentTask = currentDeviceFlow->task[currentDeviceFlow->getTaskPointer()];
369 
370 // Serial.print(currentDeviceFlow->getName());
371 // Serial.println(currentDevice->getName());
372 
373  if (currentDevice->getState() > SLEEPING) { // (== MOVING || MOVE_DONE || PENDING_MOVES)
374  if (currentDevice->getState() == MOVING) { // if device is moving...
375  currentDevice->operateTask(); // so: move on!
376 
377  switch (currentDevice->getStopEvent()) { // is there a stopEvent defined?
378  case DATE: // yes, device shall stop by date
379  if (taskTime > currentDevice->getStartTime() + currentDevice->getStartDelay() + currentDevice->getTimeout()) { // it's time to stop!
380  handleStopEvent(taskTime, currentDeviceFlow);
381  }
382  break;
383 
384  case BUTTON: // device shall stop by button
385  if (currentTask->getStopButton()->readIfActiv()) { // it's time to stop!
386  handleStopEvent(taskTime, currentDeviceFlow);
387  }
388  break;
389 
390  case FOLLOW: // device shall stop when a device reached a certain position
391  if (currentTask->getStopTriggerDevice()->getCurrentTaskID() > currentTask->getStopTriggerTaskID()) { // trigger device on trigger move?
392  handleStopEvent(taskTime, currentDeviceFlow);
393  }
394  break;
395 
396  case POSITION: // device shall stop when a device reached a certain position
397  triggerDevice_stop = currentTask->getStopTriggerDevice();
398  if (triggerDevice_stop->getCurrentTaskID() >= currentTask->getStopTriggerTaskID()) { // trigger device on trigger move?
399  if ((triggerDevice_stop->getDirectionDown() && triggerDevice_stop->getCurrentPosition() <= currentTask->getStopTriggerPosition()) || (!triggerDevice_stop->getDirectionDown() && triggerDevice_stop->getCurrentPosition() >= currentTask->getStopTriggerPosition())) { // trigger position reached?
400  handleStopEvent(taskTime, currentDeviceFlow);
401  }
402  }
403  break;
404  default:
405  break;
406  }
407  } // (currentDevice->getState() == MOVING)
408  else { // if device is stopped
409 
410  if (currentDevice->getState() == MOVE_DONE) { // finished right now?
411  currentDevice->finishTask();
412  currentDevice->disableDevice();
413 
414  if (SCHEDULER_VERBOSE & SHOW_TASK_VIEW) {
415  Serial.print(taskTime);
416  Serial.print(F(": "));
417  Serial.print(currentDeviceFlow->getName());
418  Serial.print(F(" Task "));
419  Serial.print(currentDeviceFlow->getTaskPointer());
420  Serial.println(F(" done"));
421  }
422 
423  currentDeviceFlow->increaseTaskPointer(); // go for next job!
424 
425  if (currentDeviceFlow->getTaskPointer() < currentDeviceFlow->getCountOfTasks()) { // all tasks done? no!
426  currentDevice->setState(PENDING_MOVES);
427  currentDevice->prepareTask(currentDeviceFlow->task[currentDeviceFlow->getTaskPointer()]);
428  if (SCHEDULER_VERBOSE & SHOW_TASK_VIEW) {
429  Serial.print(taskTime);
430  Serial.print(F(": "));
431  Serial.print(currentDeviceFlow->getName());
432  Serial.print(F(" prepare Task "));
433  Serial.print(currentDeviceFlow->getTaskPointer());
434  Serial.print(F(": current Position: "));
435  Serial.print(currentDevice->getCurrentPosition());
436  Serial.print(F(" target: "));
437  Serial.println(currentDevice->getTarget());
438  }
439 
440  }
441  else { // all tasks are done
442  currentDevice->setState(SLEEPING);
443  if (SCHEDULER_VERBOSE & SHOW_TASK_VIEW) {
444  Serial.print(taskTime);
445  Serial.print(F(": "));
446  Serial.print(currentDeviceFlow->getName());
447  Serial.println(F(" finished all Tasks"));
448  }
449  }
450  }
451  else { // (currentDevice->getState() == MOVE_DONE) // device is idle
452 
453 // Serial.print(taskTime);
454 // Serial.print(": ");
455 // Serial.print(currentDevice->getStartEvent() == DATE);
456 // Serial.print(": ");
457 // Serial.println(currentDevice->getStartTime());
458 //
459 
460 
461  switch (currentDevice->getStartEvent()) { // what kind of startEvent is given?
462  case DATE: // start the next move by date
463  if (taskTime > currentDevice->getStartTime()) {
464  handleStartEvent(taskTime, currentDeviceFlow);
465  }
466  break;
467 
468  case BUTTON: // start the next move by button
469  if (currentTask->getStartButton()->readIfActiv()) {
470  handleStartEvent(taskTime, currentDeviceFlow);
471  }
472  break;
473 
474  case FOLLOW: // start the next move when a device reached a certain
475  if (currentTask->getStartTriggerDevice()->getCurrentTaskID() > currentTask->getStartTriggerTaskID()) { // is the trigger servo doing the trigger move?
476 
477  handleStartEvent(taskTime, currentDeviceFlow);
478  }
479  break;
480 
481  case POSITION: // start the next move when a device reached a certain
482  triggerDevice_start = currentTask->getStartTriggerDevice();
483  if (triggerDevice_start->getCurrentTaskID() >= currentTask->getStartTriggerTaskID()) { // is the trigger servo doing the trigger move?
484 
485  if ((triggerDevice_start->getDirectionDown() && triggerDevice_start->getCurrentPosition() <= currentTask->getStartTriggerPosition()) || (!triggerDevice_start->getDirectionDown() && triggerDevice_start->getCurrentPosition() >= currentTask->getStartTriggerPosition())) {
486  // did the trigger servo pass the trigger position?
487  handleStartEvent(taskTime, currentDeviceFlow);
488  }
489  }
490  break;
491  default:
492  break;
493  }
494  }
495  }
496  }
497  ongoingOperations += currentDevice->getState(); // ongoing operations on any device?
498 
499  }
500 
501  // if (taskTime - lastPrintTime > 1000) {
502  // device[6]->getReadOut(0);
503  // lastPrintTime = taskTime;
504  // }
505 
506  if (theButtonID < currentWorkflow->getCountOfFlowControls()) {
507  currentFlowControl = currentWorkflow->flowControl[theButtonID];
508  if (currentFlowControl->button->readIfActiv()) {
509  for (unsigned char theActionID = 0; theActionID < currentFlowControl->getCountOfActions(); theActionID++) {
510  currentAction = currentFlowControl->action[theActionID];
511  if (!currentAction->getActionDone()) {
512  if ((currentAction->getValidTaskID() == -1) || (currentAction->getValidTaskID() == currentAction->targetDeviceFlow->getTaskPointer())) {
513 
514  currentWorkflow->setWorkflowInfo(currentAction->getWorkflowInfoCode());
515 
516 // notificationCode = min(notificationCode, currentAction->notificationCode);
517 // notificationText = notificationText + currentAction->notificationText + "\n";
518 
519  if (SCHEDULER_VERBOSE & SHOW_TASK_VIEW) {
520  Serial.print(taskTime);
521  Serial.print(F(": "));
522  Serial.print(currentButton->getName());
523  Serial.print(F(" rise Action "));
524  Serial.println(theActionID);
525 // Serial.print(F(" with notification: "));
526 // Serial.print(currentAction->notificationCode);
527 // Serial.print(F(" --> "));
528 // Serial.println(currentAction->notificationText);
529  }
530 
531  if (currentAction->getTargetAction() == BREAK_LOOP) {
532  breakLoop = true;
533  break;
534  }
535 
536  if (currentAction->getFollowingTaskID() > -1) {
537  currentAction->targetDeviceFlow->setTaskPointer(currentAction->getFollowingTaskID() - 1);
538  switch (currentAction->getTargetAction()) {
539  case START_NEXT_TASK:
540  handleStartEvent(taskTime, currentAction->targetDeviceFlow);
541  break;
542  case STOP_TASK_AND_SWITCH:
543  currentAction->targetDeviceFlow->device->setSwitchTaskPromptly(true);
544  case STOP_TASK:
545  currentAction->targetDeviceFlow->device->setStopping(STOP_NORMAL);
546  handleStopEvent(taskTime, currentAction->targetDeviceFlow);
547  break;
548  case STOP_TASK_SHARP_AND_SWITCH:
549  currentAction->targetDeviceFlow->device->setSwitchTaskPromptly(true);
550  case STOP_TASK_SHARP:
551  currentAction->targetDeviceFlow->device->setStopping(STOP_IMMEDIATELY);
552  handleStopEvent(taskTime, currentAction->targetDeviceFlow);
553  default:
554  break;
555  }
556  } else {
557  if (currentAction->getValidTaskID() > -1) {
558  currentWorkflow->setWorkflowInfo(WORKFLOW_CANCELED_ON_FLOWCONTROL_ERROR);
559  breakLoop = true;
560  break;
561  }
562  }
563  currentAction->setActionDone(true);
564  }
565  }
566  }
567  }
568  theButtonID++;
569  } else {
570  theButtonID = 0;
571  }
572 
573 
574  if (breakLoop) {
575  break;
576  }
577 
578 
579  loopCounter++;
580 
581 
582  } while (ongoingOperations > 0);
583 
584 
585  for (unsigned char df = 0; df < currentWorkflow->getCountOfDeviceFlows(); df++) {
586  currentDevice = currentWorkflow->deviceFlow[df]->device;
587  currentDevice->disableDevice();
588  currentDevice->detachDevice();
589  }
590 
591  if (breakLoop) {
592  Serial.println(F("workflow canceled"));
593  } else {
594  Serial.println(F("workflow ended normaly:"));
595  }
596 
597  Serial.print(F("notificationCode:"));
598  Serial.println(notificationCode);
599  Serial.print(F("notificationText:"));
600  Serial.println(notificationText);
601  Serial.println();
602  Serial.print(F("loops: "));
603  Serial.print((signed long)loopCounter);
604  Serial.print(F(", elapsed time: "));
605  Serial.print((int)((millis() - taskStartTime) / 1000));
606  Serial.print(F(", loops/second: "));
607  Serial.print(1000.0 * loopCounter / (millis() - taskStartTime));
608  Serial.println();
609  Serial.println();
610 
611  } else {
612  Serial.println(F("workflow was disabled"));
613 
614  }
615 
616 
617  return notificationCode;
618 }

621 void CCDeviceScheduler::handleStartEvent(unsigned long taskTime, CCDeviceFlow* currentDeviceFlow) {
622
