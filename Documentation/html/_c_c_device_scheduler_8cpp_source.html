461  unsigned long loopCounter = 0;
462  unsigned int ongoingOperations = 0;
463  deviceInfoCode info;
464  bool breakLoop = false;
465  notificationCode = 0;
466  notificationText = "";
467 
468 
469  if ((char)currentWorkflow->getWorkflowInfo() >= 0) {
470 
471  // prepare the loop
472 
473  if (verbosity & SHOW_TASK_VIEW) {
474  Serial.print(F("[CCDeviceScheduler]: "));
475  Serial.print(schedulerName);
476  Serial.print(F(": prepare workflow "));
477  Serial.println(currentWorkflow->getName());
478  }
479 
480 #ifdef ARDUINO_SIMULATION
481  clearDeviceDisplay();
482 #endif
483 
484  for (unsigned int df = 0; df < currentWorkflow->getCountOfDeviceFlows(); df++) {
485  currentDeviceFlow = currentWorkflow->deviceFlow[df];
486  currentDeviceFlow->setTaskPointer(0);
487 
488  currentDevice = currentDeviceFlow->getDevice();
489  currentDevice->attachDevice();
490 
491  // prepare first Tasks
492  currentDeviceFlow->setProposedTaskID(0);
493  currentDeviceFlow->handlePreparation(0);
494  currentDeviceFlow->setProposedTaskID(currentDeviceFlow->getTaskPointer() + 1);
495  currentDevice->setState(PENDING_MOVES);
496 
497  if (verbosity & SHOW_TASK_VIEW) {
498  Serial.print(currentWorkflow->getName());
499  Serial.print(F("->"));
500  Serial.print(currentDeviceFlow->getName());
501  Serial.print(F(": setup first Task: current Position: "));
502  Serial.print(currentDevice->getCurrentPosition());
503  Serial.print(F(", target: "));
504  Serial.println(currentDevice->getTarget());
505  }
506 
507 #ifdef ARDUINO_SIMULATION
508  copyDeviceName(df, currentDeviceFlow->getDevice()->getName());
509 #endif
510 
511  }
512 
513 
514  for (unsigned int fc = 0; fc < currentWorkflow->getCountOfFlowControls(); fc++) {
515  for (unsigned int a = 0; a < currentWorkflow->flowControl[fc]->getCountOfActions(); a++) {
516  currentWorkflow->flowControl[fc]->getAction(0)->setActionDone(false);
517  }
518  }
519 
520 
521  for (unsigned int c = 0; c < currentWorkflow->getCountOfControls(); c++) {
522  currentWorkflow->control[c]->read();
523  }
524 
525 
526  // start the workflow
527  if (verbosity & SHOW_TASK_VIEW) {
528  Serial.print(F("[CCDeviceScheduler]: "));
529  Serial.print(schedulerName);
530  Serial.print(F(": run workflow "));
531  Serial.println(currentWorkflow->getName());
532  }
533 
534  unsigned long loopTime;
535  unsigned long loopMicros, elapsedLoopMicros;
536  unsigned long deviceLoopPeriod_min = 100000;
537  unsigned long deviceLoopPeriod_max = 0;
538  unsigned long controlPeriod_min = 100000;
539  unsigned long controlPeriod_max = 0;
540  unsigned long flowControlPeriod_min = 100000;
541  unsigned long flowControlPeriod_max = 0;
542  unsigned long taskStartTime = millis();
543 
544  // unsigned long lastPrintTime = taskStartTime;
545 
546  unsigned int theFlowControlID = 0;
547  unsigned int theControlID = 0;
548 
549  do {
550  ongoingOperations = 0;
551 
552  loopMicros = micros();
553 
554  for (unsigned int df = 0; df < currentWorkflow->getCountOfDeviceFlows(); df++) {
555  loopTime = millis();
556  taskTime = loopTime - taskStartTime;
557 
558  currentDeviceFlow = currentWorkflow->deviceFlow[df];
559  currentDevice = currentDeviceFlow->getDevice();
560  currentTask = currentDeviceFlow->task[currentDeviceFlow->getTaskPointer()];
561 
562  if (currentDevice->getState() > SLEEPING) { // (== MOVING || MOVE_DONE || PENDING_MOVES)
563  if (currentDevice->getState() == MOVING) { // if device is moving...
564  currentDevice->operateTask(); // so: move on!
565 
566 
567 #ifdef ARDUINO_SIMULATION
568  copyDevicePosition(df, currentDevice->getCurrentPosition());
569  if (currentDevice->getType() == SERVODEVICE) {
570  copyAnalogInValue(df, 2000.0 - currentDevice->getCurrentPosition());
571  }
572  if (currentDeviceFlow->getID() == 1) {
573  copyAnalogInValue(7, -(currentDevice->getCurrentPosition()));
574 // copyAnalogInValue(10, -(currentDevice->getCurrentPosition()));
575  }
576 #endif
577 
578  switch (currentDeviceFlow->getStopEvent()) { // is there a stopEvent defined?
579  case DATE: // yes, device shall stop by date
580  if (currentDeviceFlow->isStopTimeReached(taskTime)) { // it's time to stop!
581 
582  if (verbosity == SHOW_TASK_VIEW) {
583  printMessage(taskTime, currentDeviceFlow, TRIGGER_TASK_STOP);
584  }
585 
586  currentDeviceFlow->handleStopEvent(taskTime);
587  }
588  break;
589 
590  case CONTROL: // device shall stop by button
591  if (currentDeviceFlow->isStopTargetReached()) {
592 
593  if (verbosity == SHOW_TASK_VIEW) {
594  printMessage(taskTime, currentDeviceFlow, TRIGGER_TASK_STOP);
595  }
596 
597  currentDeviceFlow->handleStopEvent(taskTime);;
598  }
599  break;
600 
601  case FOLLOW: // device shall stop when a device finished a task
602  if (currentDeviceFlow->getStopTriggerDevice()->getCurrentTaskID() > currentDeviceFlow->getStopTriggerTaskID()) { // trigger device on trigger move?
603 
604  if (verbosity == SHOW_TASK_VIEW) {
605  printMessage(taskTime, currentDeviceFlow, TRIGGER_TASK_STOP);
606  }
607 
608  currentDeviceFlow->handleStopEvent(taskTime);;
609  }
610  break;
611 
612  case FOLLOW_ME: // device shall stop when a device finished a task
613 // currentDeviceFlow->handleStopEvent(taskTime);;
614  // guess this is obsolet!!!
615  break;
616 
617  case POSITION: // device shall stop when a device reached a certain position
618  triggerDevice_stop = currentDeviceFlow->getStopTriggerDevice();
619  if (triggerDevice_stop->getCurrentTaskID() >= currentDeviceFlow->getStopTriggerTaskID()) { // trigger device on trigger move?
620  if (triggerDevice_stop->getCurrentTaskID() > currentDeviceFlow->getStopTriggerTaskID()) {
621 
622  if (verbosity == SHOW_TASK_VIEW) {
623  printMessage(taskTime, currentDeviceFlow, TRIGGER_TASK_STOP);
624  }
