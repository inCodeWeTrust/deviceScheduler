368 void CCStepperDevice::initiateStop() { // irregular stopping
369  if (currentMicroStep < microStepsForAcceleration) { // if stop request appears while accelerating
370  // stop acceleration immediately and decelerate symmetrical
371  // what is my speed?
372  // v * v [steps/s] = 2 * a * steps = 2 * a * microStep / microStepsPerFullStep
373  veloBySquare = 2 * abs(acceleration) * currentMicroStep / (1 << highestSteppingMode);
374  velocity = sqrt(veloBySquare);
375 
376  // do deceleration symetrical to acceleration: take same amount of microSteps for deceleration as used for acceleration till here
377  microStepsToGo = 2 * currentMicroStep; // stepsToGo = stepsForAcceleration + stepsForDeceleration
378  microStepsForAcceleration = currentMicroStep; // stepsForAcceleration = currentStep = stepsForDeceleration
379  microStepsForAccAndConstSpeed = currentMicroStep; // stepsForConstSpeed = 0
380  timeForAccAndConstSpeed = stepExpiration; // time for start of deceleration is time of next step
381  return;
382  }
383  if (currentMicroStep < microStepsForAccAndConstSpeed) { // when going with constant speed
384  microStepsToGo = currentMicroStep + (stepsForDeceleration << highestSteppingMode); // use nominated deceration values
385  microStepsForAccAndConstSpeed = currentMicroStep; // no more stepsForConstSpeed
386  timeForAccAndConstSpeed = stepExpiration; // time for start of deceleration is time of next step
387  return;
388  }
