96 void CCDevice::deleteTasks() {
97  if (CCDEVICE_VERBOSE & CCDEVICE_BASICOUTPUT) {
98  Serial.print(F("[CCDevice]: delete tasks of "));
99  Serial.print(deviceName);
100  Serial.print(F(": move "));
101  }
102  for (int j = countOfTasks - 1; j >= 0; j--) {
103  if (CCDEVICE_VERBOSE & CCDEVICE_BASICOUTPUT) {
104  Serial.print(F(" #"));
105  Serial.print(j);
106  }
107  delete task[j];
108  task[j] = NULL;
109  }
110  if (CCDEVICE_VERBOSE & CCDEVICE_BASICOUTPUT) {
111  Serial.println();
112  Serial.println();
113  }
114 
115  state = SLEEPING;
116  taskPointer = 0;
117  countOfTasks = 0;
118 
119 }
