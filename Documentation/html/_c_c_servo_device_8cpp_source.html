158  // wayToGo / 2 = v * v / a ==> a = 2 * v * v / wayToGo
159  acceleration = 2 * velocity * velocity / wayToGo;
160  }
161 
162  // v^2 = 2 * a * s --> s = v^2 / (2 * a) = 0.5 * v * v / a
163  wayForAcceleration = (signed long) (0.5 * velocity * velocity / acceleration);
164

172

328 void CCServoDevice::operateTask() {
329  lastCycleTime = elapsedTime;
330  elapsedTime = millis() - t0;
331 
332  if (stopping == STOP_DYNAMIC) {
333  if (dynamicalStop == false) {
334  sensorValue = approximationControl->value();
335  if ((sensorValue > initiatePerformanceValue && (!sensorValuesFalling)) || (sensorValue < initiatePerformanceValue && sensorValuesFalling)) {
336  timeForAcceleration = elapsedTime;
337  timeForConstantSpeed = 0;
338  lastCycleTime = 0;
339  c_perform = 1.0 / (initiatePerformanceValue - approximationTarget);
340  dynamicalStop = true;
341  }
342  }
343  }
344 
345  // ramp up
346  if (elapsedTime < (signed long)timeForAcceleration) {
347  // s = s0 + 0.5 * a * t^2 * (1000ms / sec)^2
348  deltaS = elapsedTime * elapsedTime * acceleration / 2000000;
349  currentPosition = startPosition + deltaS;
350  theServo.writeMicroseconds(currentPosition);
351 
352  if (verbosity & MOVEMENTDEBUG) {
353  Serial.print(F("[CCServoDevice]: "));
354  Serial.print(deviceName);
355  Serial.print(F(": ramp up: cur: "));
356  Serial.print((int)currentPosition);
357  Serial.print(F(", start: "));
358  Serial.print((int)startPosition);
359  Serial.print(F(", deltaS: "));
360  Serial.println((long)deltaS);
361  }
362 
363  return;
364  }
365 
366  // go with constant speed
367  elapsedTime -= timeForAcceleration;
368  if (elapsedTime < (signed long)timeForConstantSpeed) {
369  // s = s0 + s1 + deltaS = s0 + s1 + v * t
370  deltaS = elapsedTime * velocity / 1000;
371  currentPosition = startPosition + wayForAcceleration + deltaS;
372  theServo.writeMicroseconds(currentPosition);
373 
374  if (verbosity & MOVEMENTDEBUG) {
375  Serial.print(F("[CCServoDevice]: "));
376  Serial.print(deviceName);
377  Serial.print(F(": go constant: cur: "));
378  Serial.print((int)currentPosition);
379  Serial.print(F(", start: "));
380  Serial.print((int)startPosition);
381  Serial.print(F(", wayAcc: "));
382  Serial.print((int)wayForAcceleration);
383  Serial.print(F(", deltaS: "));
384  Serial.println((long)deltaS);
385  }
386 
387  return;
388  }
389  // stop dynamically?
390  if (dynamicalStop) {
391  deltaDeltaNorm = (float)(elapsedTime - lastCycleTime) * velocity / 1000.0;
392 
393  sensorValue = approximationControl->value();
394 
395  performanceFactor = c_perform * (sensorValue - approximationTarget);
396 
397  // int dif = targetValue - sensorValue;
398  //
399  // if (dif < 0) {
400  // currentPosition += deltaDeltaNorm * (1 - 1 / dif);
401  // }
402  // if (dif > 0) {
403  // currentPosition += deltaDeltaNorm * (1 - 1 / dif);
404  // }
405  //
406  // if (performanceFactor > 0) {
407  // performanceFactor = pow(performanceFactor, stopPerformance);
408  // } else {
409  // performanceFactor = -pow(fabs(performanceFactor), stopPerformance);
410  // }
411 
412  currentPosition += deltaDeltaNorm * performanceFactor;
413  currentPosition = fmin(currentPosition, maxPosition);
414  currentPosition = fmax(currentPosition, minPosition);
415 
416  theServo.writeMicroseconds(currentPosition);
417 
418  if (verbosity & MOVEMENTDEBUG) {
419  Serial.print("elapsedTime: ");
420  Serial.print(elapsedTime);
421  Serial.print(", performanceFactor: ");
422  Serial.print(performanceFactor);
423  Serial.print(", sensorValue: ");
424  Serial.print(sensorValue);
425  Serial.print(", deltaDeltaNorm: ");
426  Serial.print(deltaDeltaNorm);
427  Serial.print(", deltaS: ");
428  Serial.print(deltaS);
429  Serial.print(", currentPosition: ");
430  Serial.println(currentPosition);
431  }
432 
433 
434  if (approximation == SKIP_APPROXIMATION_NEVER) {
435  return;
436  } else {
437  if (fabs(sensorValue - approximationTarget) < sensorTreshold) {
438  if (valueCounter++ < approximation) {
439  return;
440  }
441  } else {
442  valueCounter = 0;
443  return;
444  }
445  }
446 
447 
448  if (verbosity & BASICOUTPUT) {
449  Serial.print(F("[CCServoDevice]: "));
450  Serial.print(deviceName);
451  Serial.print(F(": stopped at sensor: "));
452  Serial.println(sensorValue);
453  }
454  targetPosition = currentPosition;
455 
456  } else {
457  // generic ramp down
458  elapsedTime -= timeForConstantSpeed;
459  if (elapsedTime < (signed long)timeForAcceleration) {
460  // t^2 = (timeForAcceleration - elapsedTime)^2 = (elapsedTime - timeForAcceleration)^2;
461  elapsedTime -= timeForAcceleration;
462  // s = sges - deltaS = 0.5 * a * t^2 = 0.5 * a * (time_in_millis / 1000)^2 = 1/2 * 1/1000000 * a * t^2
463  deltaS = elapsedTime * elapsedTime * acceleration / 2000000;
464  currentPosition = targetPosition - deltaS;
465  theServo.writeMicroseconds(currentPosition);
466 
467  if (verbosity & MOVEMENTDEBUG) {
468  Serial.print(F("[CCServoDevice]: "));
469  Serial.print(deviceName);
470  Serial.print(F(": ramp down: cur: "));
471  Serial.print((int)currentPosition);
472  Serial.print(F(", target: "));
473  Serial.print((int)targetPosition);
474  Serial.print(F(", deltaS: "));
475  Serial.println((long)deltaS);
476  }
477 
478  return;
479  }
480  }
481 
482 
483  // if we didnt return up to here, we are done!
484 
485  theServo.writeMicroseconds(targetPosition);
486  currentPosition = targetPosition;
487 
488  stopTask();
489
